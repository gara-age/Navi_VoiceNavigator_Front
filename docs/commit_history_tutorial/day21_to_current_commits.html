2. 수동 닫기와 표시 설정 반영

토스트는 자동으로 닫히는 것만으로 끝나지 않습니다. 사용자가 직접 닫을 수 있는 X 버튼과, 설정 모달의 DisplaySettings에 따른 큰 글씨/다크/고대비 대응이 함께 들어갑니다.

class _ToastOverlay {
  OverlayEntry? _entry;
  Timer? _dismissTimer;

  void dismiss() {
    _dismissTimer?.cancel();
    _dismissTimer = null;
    _entry?.remove();
    _entry = null;
  }

  void show(
    BuildContext context, {
    required String title,
    required String message,
    required AppToastState state,
    required DisplaySettings? displaySettings,
    required int durationMs,
  }) {
    dismiss();
    _entry = OverlayEntry(
      builder: (context) => _ToastView(
        title: title,
        message: message,
        state: state,
        displaySettings: displaySettings,
        onClose: dismiss,
      ),
    );
    Overlay.of(context).insert(_entry!);
    _dismissTimer = Timer(Duration(milliseconds: durationMs), dismiss);
  }
}

1. Flutter에서 플랫폼 팝업을 호출하는 공통 서비스 추가

lib/shared/services/taskbar_popup_service.dart
enum TaskbarPopupState {
  info,
  listening,
  processing,
  success,
  warning,
  error,
}

enum TaskbarPopupThemeMode {
  light,
  dark,
  contrast,
}
class TaskbarPopupService {
  static const MethodChannel _macToastChannel =
      MethodChannel('voice_navigator/taskbar_popup');

  Future<bool> show({
    required String title,
    required String message,
    required TaskbarPopupState state,
    required TaskbarPopupThemeMode themeMode,
    required bool largeText,
    int durationMs = 5000,
  }) async {
    if (Platform.isMacOS) {
      final shown = await _macToastChannel.invokeMethod<bool>(
        'showMacToast',
        {
          'title': title,
          'message': message,
          'durationMs': durationMs,
          'state': state.name,
          'themeMode': themeMode.name,
          'largeText': largeText,
        },
      );
      return shown ?? true;
    }

    if (!Platform.isWindows) {
      return false;
    }

    final scriptPath = _findScriptPath();
    if (scriptPath == null) {
      return false;
    }

    await Process.start(
      'powershell',
      [
        '-NoProfile',
        '-NonInteractive',
        '-WindowStyle', 'Hidden',
        '-ExecutionPolicy', 'Bypass',
        '-File', scriptPath,
        '-Title', title,
        '-Message', message,
        '-DurationMs', durationMs.toString(),
        '-State', state.name,
        '-ThemeMode', themeMode.name,
        '-LargeText', largeText ? '1' : '0',
      ],
    );

    return true;
  }
}

2. Windows에서는 fallback 토스트를 막고, 다른 플랫폼만 Flutter fallback 허용

lib/features/notifications/presentation/app_toast.dart

이 부분이 중요합니다. Windows에서는 taskbar popup이 목적이므로, 스크립트 호출에 실패했다고 해서 다시 앱 내부 토스트를 띄우지 않습니다.

Future<void> _showTaskbarPopupOrFallback(
  BuildContext context, {
  required String title,
  required String message,
  required AppToastState state,
  DisplaySettings? displaySettings,
  required int durationMs,
}) async {
  final shown = await TaskbarPopupService.instance.show(
    title: title,
    message: message.trim(),
    durationMs: durationMs,
    state: _mapToastState(state),
    themeMode: _resolveThemeModeFromDisplay(displaySettings),
    largeText: displaySettings?.largeText ?? false,
  );

  if (!shown && !Platform.isWindows && context.mounted) {
    _ToastOverlay.instance.show(
      context,
      title: title,
      message: message,
      state: state,
      displaySettings: displaySettings,
      durationMs: durationMs,
    );
  }
}

3. Windows 실행 파일 옆에서도 스크립트를 찾도록 경로 탐색 강화

lib/shared/services/taskbar_popup_service.dart

VOICE_NAVIGATOR_ROOT만 믿으면 Release 폴더에서 .exe 직접 실행 시 스크립트를 못 찾습니다. 그래서 현재 작업 디렉터리와 실행 파일 기준 상위 폴더까지 모두 탐색합니다.

String? _findScriptPath() {
  final envRoot = Platform.environment['VOICE_NAVIGATOR_ROOT'];
  if (envRoot != null && envRoot.isNotEmpty) {
    final fromEnv =
        '$envRoot${Platform.pathSeparator}scripts${Platform.pathSeparator}show_taskbar_popup.ps1';
    if (File(fromEnv).existsSync()) {
      return fromEnv;
    }
  }

  final roots = <String>{};

  void addAncestors(String startPath) {
    var dir = Directory(startPath);
    for (var i = 0; i < 8; i++) {
      roots.add(dir.path);
      final parent = dir.parent;
      if (parent.path == dir.path) break;
      dir = parent;
    }
  }

  addAncestors(Directory.current.path);
  addAncestors(File(Platform.resolvedExecutable).parent.path);

  for (final root in roots) {
    final candidate =
        '$root${Platform.pathSeparator}scripts${Platform.pathSeparator}show_taskbar_popup.ps1';
    if (File(candidate).existsSync()) {
      return candidate;
    }
  }

  return null;
}

4. macOS에서는 MethodChannel + NSPanel

macos/Runner/AppDelegate.swift
func configureToastChannel(messenger: FlutterBinaryMessenger) {
  let channel = FlutterMethodChannel(
    name: "voice_navigator/taskbar_popup",
    binaryMessenger: messenger
  )

  channel.setMethodCallHandler { [weak self] call, result in
    guard call.method == "showMacToast" else {
      result(FlutterMethodNotImplemented)
      return
    }

    guard let args = call.arguments as? [String: Any] else {
      result(false)
      return
    }

    self?.showMacToast(args: args)
    result(true)
  }
}
private func showMacToast(args: [String: Any]) {
  let largeText = args["largeText"] as? Bool ?? false
  let width: CGFloat = largeText ? 372 : 320
  let height: CGFloat = largeText ? 118 : 96

  let panel = NSPanel(
    contentRect: NSRect(x: x, y: y, width: width, height: height),
    styleMask: [.nonactivatingPanel, .borderless],
    backing: .buffered,
    defer: false
  )
  panel.isFloatingPanel = true
  panel.level = .statusBar
  panel.backgroundColor = .clear
  panel.isOpaque = false
  panel.collectionBehavior = [.canJoinAllSpaces, .fullScreenAuxiliary]
}

macOS는 SwiftUI 카드형 뷰를 패널에 얹어 현재 프로젝트 카드 디자인을 그대로 유지합니다.

3. 포커스를 덜 뺏는 no-activate 창

Add-Type @"
public static class PopupWindowInterop {
  public const int GWL_EXSTYLE = -20;
  public const int WS_EX_TOOLWINDOW = 0x00000080;
  public const int WS_EX_NOACTIVATE = 0x08000000;
}
"@

$window.Add_SourceInitialized({
  $helper = New-Object System.Windows.Interop.WindowInteropHelper($window)
  $extendedStyle = [PopupWindowInterop]::GetWindowLong($helper.Handle, [PopupWindowInterop]::GWL_EXSTYLE)
  $extendedStyle = $extendedStyle -bor [PopupWindowInterop]::WS_EX_TOOLWINDOW -bor [PopupWindowInterop]::WS_EX_NOACTIVATE
  [PopupWindowInterop]::SetWindowLong($helper.Handle, [PopupWindowInterop]::GWL_EXSTYLE, $extendedStyle) | Out-Null
})

여기서 WS_EX_NOACTIVATE를 넣어야 팝업이 뜰 때 메인 앱 포커스를 빼앗는 현상을 줄일 수 있습니다.

4. Pretendard와 SVG 아이콘을 스크립트가 직접 읽음

function Get-PretendardFontFamily { ... }
function New-SvgViewbox { ... }
function Get-IconPath {
  param([string]$PopupState)
  $name = switch ($PopupState.ToLowerInvariant()) {
    'listening' { 'mic.svg' }
    'processing' { 'hourglass.svg' }
    'success' { 'check.svg' }
    'warning' { 'warning.svg' }
    'error' { 'error.svg' }
    default { 'info.svg' }
  }
  return Get-ScriptAssetPath "popup_icons\$name"
}

즉 Windows 팝업은 Flutter 아이콘을 재사용하지 않고, 별도 SVG 자산을 WPF shape로 변환해 그립니다.

현재 최종 상태

지금 시점의 팝업 시스템은 “공통 Dart 진입점 + OS별 렌더러 + 플랫폼별 fallback” 구조입니다.

최종 흐름

  1. showAppToast(...)가 호출된다.
  2. TaskbarPopupService.instance.show(...)가 먼저 실행된다.
  3. macOS면 MethodChannel이 AppDelegate.swift로 전달된다.
  4. Windows면 show_taskbar_popup.ps1가 WPF 팝업을 띄운다.
  5. Windows가 아니고 OS 팝업이 실패한 경우에만 Flutter overlay 토스트가 fallback으로 뜬다.
